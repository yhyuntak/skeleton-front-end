# 프로젝트 구조
```
src/
├── assets/          # 정적 파일 (이미지, 폰트, CSS 등)
├── components/      # 재사용 가능한 컴포넌트
├── layouts/         # 페이지 레이아웃 컴포넌트
├── pages/           # 라우팅되는 페이지 컴포넌트
├── hooks/           # 커스텀 React Hooks
├── store/           # 상태 관리 (Redux, Context API 등)
├── styles/          # 글로벌 스타일 파일 (Tailwind config 포함)
├── utils/           # 유틸리티 함수
├── services/        # API 통신 로직 (Axios 관련 파일)
├── types/           # 타입 정의 파일
├── App.tsx          # 메인 App 컴포넌트
├── index.tsx        # ReactDOM 렌더링 진입점
└── main.tsx         # 프로젝트 진입점 (Vite 사용 시)

```
